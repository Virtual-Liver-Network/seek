def delete_rdf_file
  public_path = public_rdf_storage_path
  private_path = private_rdf_storage_path
  FileUtils.rm(public_path) if File.exists?(public_path)
  FileUtils.rm(private_path) if File.exists?(private_path)
end

def rdf_storage_path inner_dir=nil?
  inner_dir ||= self.can_view?(nil) ? "public" : "private"
  path = File.join(Seek::Config.rdf_filestore_path,inner_dir)
  if !File.exists?(path)
    FileUtils.mkdir_p(path)
  end

  filename = rdf_storage_filename
  File.join(path,filename)
end

    saves the RDF to a file according to the configured filestore, in a directory rdf/. Filenames are based upon the asset type, the Rails.env, and the asset id. rdf for private asset (not publically visible) are stored in a private subdirectory, and those that are public are stored in a public subdirectory

def save_rdf_file
  delete_rdf_file
  path = self.rdf_storage_path

  File.open(path,"w") do |f|
    f.write(self.to_rdf)
    f.flush
  end
  path
end
